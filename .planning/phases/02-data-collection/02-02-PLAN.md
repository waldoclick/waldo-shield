   ```python
   from datetime import datetime, timedelta, timezone
   from cloudflare import Cloudflare
   
   class CloudflareClient:
       """Wrapper for Cloudflare API operations."""
       
       def __init__(self, token: str):
           self._client = Cloudflare(api_token=token, max_retries=2)
       
       def get_security_events(self, zone_id: str, hours: int = 24) -> dict:
           """Retrieve WAF/firewall events for the last N hours."""
           # GraphQL query for firewallEventsAdaptive
           # Return: {zone_id, period_hours, total_events, events: [...], by_action: {...}, by_source: {...}}
       
       def get_traffic_analytics(self, zone_id: str, hours: int = 24) -> dict:
           """Retrieve traffic analytics (requests, cached, blocked)."""
           # GraphQL query for httpRequestsAdaptiveGroups + firewallEventsAdaptiveGroups
           # Return: {zone_id, period_hours, total_requests, cached_requests, blocked_requests, blocked_percentage}
       
       def get_rate_limit_rules(self, zone_id: str) -> list:
           """Retrieve configured rate limiting rules."""
           # Use rulesets.list(zone_id=zone_id), filter for http_ratelimit phase
           # Return: [{id, expression, action, period, requests_per_period}, ...]
   ```

4. Implement GraphQL queries per research patterns:
   - Use ISO format for datetime: (datetime.now(timezone.utc) - timedelta(hours=hours)).isoformat()
   - Handle empty responses gracefully
   - Catch exceptions and return {error: str} instead of raising

5. Add helper function:
   ```python
   def collect_cloudflare_data(token: str, zone_id: str) -> dict:
       """Collect all Cloudflare data for a zone. Single entry point for Phase 3."""
       client = CloudflareClient(token)
       return {
           "security_events": client.get_security_events(zone_id),
           "traffic_analytics": client.get_traffic_analytics(zone_id),
           "rate_limit_rules": client.get_rate_limit_rules(zone_id),
       }
   ```
